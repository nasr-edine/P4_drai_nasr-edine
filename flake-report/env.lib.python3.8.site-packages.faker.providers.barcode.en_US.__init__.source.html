113

122 |EnUsBarcodeProvider.upc_e| for more details.

128 if number_system_digit not in [0, 1]:
129 number_system_digit = self.random_int(0, 1)
130  
131 if base[-1] <= 2:
132 code = base[:2] + base[-1:] + [0] * 4 + base[2:-1]
133 elif base[-1] <= 4:
134 code = base[:base[-1]] + [0] * 5 + base[base[-1]:-1]
135 else:
136 code = base[:5] + [0] * 4 + base[-1:]
