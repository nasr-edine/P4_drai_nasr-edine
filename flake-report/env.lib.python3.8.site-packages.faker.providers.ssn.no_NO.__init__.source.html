7 def checksum(digits, scale):
8 """
9 Calculate checksum of Norwegian personal identity code.
10  
11 Checksum is calculated with "Module 11" method using a scale.
12 The digits of the personal code are multiplied by the corresponding
13 number in the scale and summed;
14 if remainder of module 11 of the sum is less than 10, checksum is the
15 remainder.
16 If remainder is 0, the checksum is 0.
17  
18 https://no.wikipedia.org/wiki/F%C3%B8dselsnummer
19 """
20 chk_nbr = 11 - (sum(map(operator.mul, digits, scale)) % 11)
21 if chk_nbr == 11:
22 return 0
23 return chk_nbr
