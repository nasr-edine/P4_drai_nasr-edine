92 def license_plate_unified(self):
93 """Generate a unified license plate."""
94 temp = re.sub(r'\?',
95 lambda x: self.random_element(
96 self.license_plate_new_format_suffix_letters),
97 self.license_formats[0])
98 return self.numerify(temp)

100 def license_plate_by_province(self, province_prefix=None):
101 """Generate a provincial license plate.
102  
103 If a value for ``province_prefix`` is provided, the value will be used
104 as the prefix regardless of validity. If ``None``, then a valid prefix
105 will be selected at random.
106 """
107 province_prefix = province_prefix if province_prefix is not None else \
108 self.random_element(self.province_prefix)
109 temp = re.sub(r'\?',
110 lambda x: self.random_element(
111 self.license_plate_old_format_suffix_letters),
112 "#### ??")
113 return province_prefix + " " + self.numerify(temp)
