333 def _postcode_replace(self, postal_code_format):
334 """
335 Replaces all question mark ('?') occurrences with a random letter
336 from given postal_code_format, then passes result to numerify to insert
337 numbers
338 """
339 temp = re.sub(r'\?',
340 lambda x: self.postal_code_letter(),
341 postal_code_format)
342 return self.numerify(temp)
