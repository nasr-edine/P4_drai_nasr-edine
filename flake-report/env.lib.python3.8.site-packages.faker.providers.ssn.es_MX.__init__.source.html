106 def _reduce_digits(number):
107 """
108 Sum of digits of a number until sum becomes single digit.
109  
110 Example:
111 658 => 6 + 5 + 8 = 19 => 1 + 9 = 10 => 1
112 """
113 if number == 0:
114 return 0
115 if number % 9 == 0:
116 return 9
117  
118 return number % 9

121 def ssn_checksum(digits):
122 """
123 Calculate the checksum for the mexican SSN (IMSS).
124 """
125 return -sum(
126 _reduce_digits(n * (i % 2 + 1))
127 for i, n in enumerate(digits)
128 ) % 10

131 def curp_checksum(characters):
132 """
133 Calculate the checksum for the mexican CURP.
134 """
135 start = 18
136 return -sum(
137 (start - i) * CURP_CHARACTERS.index(n)
138 for i, n in enumerate(characters)
139 ) % 10

149 """
150 Mexican Social Security Number, as given by IMSS.

152 :return: a random Mexican SSN
153 """
154 office = self.random_int(min=1, max=99)
155 birth_year = self.random_int(min=0, max=99)
156 start_year = self.random_int(min=0, max=99)
157 serial = self.random_int(min=1, max=9999)
158  
159 num = f'{office:02d}{start_year:02d}{birth_year:02d}{serial:04d}'
160  
161 check = ssn_checksum(map(int, num))
162 num += str(check)
163  
164 return num
