59 # Standard conversion support for thai_strftime()
60 def _std_strftime(dt_obj: datetime, fmt_char: str) -> str:
61 """
62 Standard datetime.strftime() with normalization and exception handling.
63 """
64 str_ = ""
65 try:
66 str_ = dt_obj.strftime(f'%{fmt_char}')
67 if not str_ or str_ == f'%{fmt_char}':
68 # normalize outputs for unsupported directives
69 # in different platforms
70 # "%Q" may result "%Q", "Q", or "", make it "Q"
71 str_ = fmt_char
72 except ValueError as err: # pragma: no cover
73 # Unsupported directives may raise ValueError on Windows,
74 # in that case just use the fmt_char
75 warnings.warn(
76 (
77 f"String format directive unknown/not support: %{fmt_char}"
78 f"The system raises this ValueError: {err}"
79 ),
80 UserWarning,
81 )
82 str_ = fmt_char
83 return str_

87 def _thai_strftime(
88 dt_obj: datetime, fmt_char: str, buddhist_era: bool = True,
89 ) -> str:
90 """
91 Conversion support for thai_strftime().
92  
93 The fmt_char should be in _NEED_L10N when call this function.
94 """
95 str_ = ""
96 year = dt_obj.year
97 if buddhist_era:
98 year = year + _BE_AD_DIFFERENCE
99  
100 if fmt_char == "A":
101 # National representation of the full weekday name
102 str_ = _TH_FULL_WEEKDAYS[dt_obj.weekday()]
103 elif fmt_char == "a":
104 # National representation of the abbreviated weekday
105 str_ = _TH_ABBR_WEEKDAYS[dt_obj.weekday()]
106 elif fmt_char == "B":
107 # National representation of the full month name
108 str_ = _TH_FULL_MONTHS[dt_obj.month - 1]
109 elif fmt_char == "b":
110 # National representation of the abbreviated month name
111 str_ = _TH_ABBR_MONTHS[dt_obj.month - 1]
112 elif fmt_char == "C":
113 # Thai Buddhist century (AD+543)/100 + 1 as decimal number;
114 str_ = str(int(year / 100) + 1).zfill(2)
115 elif fmt_char == "c":
116 # Locale’s appropriate date and time representation
117 # Wed 6 Oct 01:40:00 1976
118 # พ 6 ต.ค. 01:40:00 2519 <-- left-aligned weekday, right-aligned day
119 str_ = (f'{_TH_ABBR_WEEKDAYS[dt_obj.weekday()]:<2} {dt_obj.day:>2} '
  • E501 Line too long (83 > 79 characters)
120 f'{_TH_ABBR_MONTHS[dt_obj.month - 1]} {dt_obj:%H:%M:%S} {year:04}')
121 elif fmt_char == "D":
122 # Equivalent to ``%m/%d/%y''
123 str_ = f'{dt_obj:%m/%d}/{year % 100:02}'
124 elif fmt_char == "F":
125 # Equivalent to ``%Y-%m-%d''
126 str_ = f'{year:04}-{dt_obj:%m-%d}'
127 elif fmt_char == "G":
128 # ISO 8601 year with century representing the year that contains
129 # the greater part of the ISO week (%V). Monday as the first day
130 # of the week.
131 year_G = int(dt_obj.strftime("%G"))
132 if buddhist_era:
133 year_G = year_G + _BE_AD_DIFFERENCE
134 str_ = f'{year_G:04}'
135 elif fmt_char == "g":
136 # Same year as in ``%G'',
137 # but as a decimal number without century (00-99).
138 year_G = int(dt_obj.strftime("%G"))
139 if buddhist_era:
140 year_G = year_G + _BE_AD_DIFFERENCE
141 str_ = f'{year_G % 100:02}'
142 elif fmt_char == "v":
143 # BSD extension, ' 6-Oct-1976'
144 str_ = f'{dt_obj.day:>2}-{_TH_ABBR_MONTHS[dt_obj.month - 1]}-{year:04}'
145 elif fmt_char == "X":
146 # Locale’s appropriate time representation.
147 str_ = f'{dt_obj:%H:%M:%S}'
148 elif fmt_char == "x":
149 # Locale’s appropriate date representation.
150 str_ = f'{dt_obj:%d/%m}/{year:04}'
151 elif fmt_char == "Y":
152 # Year with century
153 str_ = f'{year:04}'
154 elif fmt_char == "y":
155 # Year without century
156 str_ = f'{year % 100:02}'
157 elif fmt_char == "+":
158 # National representation of the date and time
159 # (the format is similar to that produced by date(1))
160 # Wed 6 Oct 1976 01:40:00
161 str_ = (f'{_TH_ABBR_WEEKDAYS[dt_obj.weekday()]:<2} {dt_obj.day:>2} '
  • E501 Line too long (80 > 79 characters)
162 f'{_TH_ABBR_MONTHS[dt_obj.month - 1]} {year} {dt_obj:%H:%M:%S}')
163  
164 return str_

167 def thai_strftime(
168 dt_obj: datetime,
169 fmt: str = "%-d %b %Y",
170 thai_digit: bool = False,
171 buddhist_era: bool = True,
172 ) -> str:
173 """
174 Convert :class:`datetime.datetime` into Thai date and time format.
175  
176 The formatting directives are similar to :func:`datatime.strrftime`.
177  
178 This function uses Thai names and Thai Buddhist Era for these directives:
179 * **%a** - abbreviated weekday name
180 (i.e. "จ", "อ", "พ", "พฤ", "ศ", "ส", "อา")
181 * **%A** - full weekday name
182 (i.e. "วันจันทร์", "วันอังคาร", "วันเสาร์", "วันอาทิตย์")
183 * **%b** - abbreviated month name
184 (i.e. "ม.ค.", "ก.พ.", "มี.ค.", "เม.ย.", "พ.ค.", "มิ.ย.", "ธ.ค.")
185 * **%B** - full month name
186 (i.e. "มกราคม", "กุมภาพันธ์", "พฤศจิกายน", "ธันวาคม",)
187 * **%y** - year without century (i.e. "56", "10")
188 * **%Y** - year with century (i.e. "2556", "2410")
189 * **%c** - date and time representation
190 (i.e. "พ 6 ต.ค. 01:40:00 2519")
191 * **%v** - short date representation
192 (i.e. " 6-ม.ค.-2562", "27-ก.พ.-2555")
193 """
194 thaidate_parts = []
195  
196 i = 0
197 fmt_len = len(fmt)
198 while i < fmt_len:
199 str_ = ""
200 if fmt[i] == "%":
201 j = i + 1
202 if j < fmt_len:
203 fmt_char = fmt[j]
204 if fmt_char in _NEED_L10N: # requires localization?
205 str_ = _thai_strftime(dt_obj, fmt_char, buddhist_era)
206 elif fmt_char in _EXTENSIONS:
207 fmt_char_ext = fmt_char
208 k = j + 1
209 if k < fmt_len:
210 fmt_char = fmt[k]
211 if fmt_char in _NEED_L10N:
212 str_ = _thai_strftime(
213 dt_obj, fmt_char, buddhist_era,
214 )
215 else:
216 str_ = _std_strftime(dt_obj, fmt_char)
217  
218 if fmt_char_ext == "-":
219 # GNU libc extension,
220 # no padding
221 if str_[0] and str_[0] in " 0":
222 str_ = str_[1:]
223 elif fmt_char_ext == "_":
224 # GNU libc extension,
225 # explicitly specify space (" ") for padding
226 if str_[0] and str_[0] == "0":
227 str_ = " " + str_[1:]
228 elif fmt_char_ext == "0":
229 # GNU libc extension,
230 # explicitly specify zero ("0") for padding
231 if str_[0] and str_[0] == " ":
232 str_ = "0" + str_[1:]
233 elif fmt_char_ext == "^":
234 # GNU libc extension,
235 # convert to upper case
236 str_ = str_.upper()
237 elif fmt_char_ext == "#":
238 # GNU libc extension,
239 # swap case - useful for %Z
240 str_ = str_.swapcase()
241 elif fmt_char_ext == "E":
242 # POSIX extension,
243 # uses the locale's alternative representation
244 # Not implemented yet
245 pass
246 elif fmt_char_ext == "O":
247 # POSIX extension,
248 # uses the locale's alternative numeric symbols
249 str_ = str_.translate(_HA_TH_DIGITS)
250 i = i + 1 # consume char after format char
251 else:
252 # format char at string's end has no meaning
253 str_ = fmt_char_ext
254 else: # not in _NEED_L10N nor _EXTENSIONS
255 # no known localization available, use Python's default
256 str_ = _std_strftime(dt_obj, fmt_char)
257  
258 i = i + 1 # consume char after "%"
259 else:
260 # % char at string's end has no meaning
261 str_ = "%"
262 else:
263 str_ = fmt[i]
264  
265 thaidate_parts.append(str_)
266 i = i + 1
267  
268 thaidate_text = "".join(thaidate_parts)
269  
270 if thai_digit:
271 thaidate_text = thaidate_text.translate(_HA_TH_DIGITS)
272  
273 return thaidate_text
